删除记录
delete from 表名[where 条件]
delete from student where sid=201610515101;
-->如果没有where指定条件 会将表中数据一条一条全部删除掉
-->面试问题：
-->delete 删除数据和truncate删除数据有什么区别
-->delete：一条一条删除表中的数据（数据比较少时候高效）
      truncate：先删除表再重建表（数据多的时候高效）

更新表记录
update 表名 set 列名=列值，列名2=列值2 [where 条件]
update student set sname='卢奕良' where sid=201610515101;
-->如果参数是字符串，日期要加上单引号

--多表之间的关系：
外键约束：foreign key
alter table 表名 add foreign key(cno) references 另一个表 (cid);
--cno为种类的数字代表
--cid为种类的id
--references 联系，涉及，关联 

一对多的时候在“多的表”上边添加一个外键指向“1”的表的主键
多对多的时候新建一张中间表，中间表建立外键分别两张表的主键


内链接：用外键过滤查找con=cid的
--隐式内链接
   select * from product as p,category as c on p.con=c.cid
--显示内链接--inner join
    select * from product p inner join category c p.con=c.cid

左外连接--left outer join
select...from...left outer join...on...
右外连接--right outer join
select...from...right outer join...on...

 
分页查询--limit
例如：每页数据8条，起始索引为0
          则第一页0开始第二页8开始
--第一页： select * from product limit 0,8;
--第二页： select * from product limit 8,8;

子查询：
例如查看分类名称为手机数码的所有商品
第一步--先查看分类名为手机数码的id
select cid from category where cname='手机数码';
	假如查询出来手机数码的id为1
	商品表里面的外键约束为con连接的种类表里面的cid
	则有第二步：
select * from product where con=1;
综上分析得到综合步骤：
select * from product where con=(select cid from category where cname='手机数码');
